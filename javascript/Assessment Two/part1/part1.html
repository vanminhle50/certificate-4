function sequentialSearch(inputArray, number) {
    for (let i = 0; i < inputArray.length; i++) { // Loop through the array
        if (inputArray[i] === number) { // Check if the current element matches the number
            return i; // Return the index if found
        }
    }

function binarySearch(inputArray, number) {
    let begin = 0, end = inputArray.length - 1;

    // Perform binary search
    while (begin <= end) {
        let middle = Math.floor((begin + end) / 2); // Find the middle index
        if (inputArray[middle] === number) {
            return middle; // Return the index if found
        } else if (inputArray[middle] < number) {
            begin = middle + 1; // Move to the right half if the number is larger
        } else {
            end = middle - 1; // Move to the left half if the number is smaller
        }
    }
    return -1; // Return -1 if the number is not found
}
